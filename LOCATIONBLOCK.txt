    <script>
      function blockPage() {
        document.title = "Page not found";

        const blocker = document.createElement("div");
        blocker.innerHTML = `
        <div style="
          position:fixed;
          top:0;
          left:0;
          width:100vw;
          height:100vh;
          display:flex;
          align-items:center;
          justify-content:center;
          background:#f3f4f6;
          z-index:999999;
          font-family:sans-serif;
        ">
          <div style="
            background:#fff;
            padding:32px;
            border-radius:16px;
            box-shadow:0 10px 30px rgba(0,0,0,0.1);
            text-align:center;
            max-width:400px;
            width:90%;
          ">
            <h1 style="font-size:64px;font-weight:900;color:#dc2626;margin-bottom:8px;">
              404
            </h1>

            <h2 style="font-size:24px;font-weight:700;margin-bottom:8px;">
              Page not found
            </h2>

          
          </div>
        </div>
      `;

        document.body.appendChild(blocker);

        // hide Zoho if exists
        const style = document.createElement("style");
        style.innerHTML = `
        #zsiq_float, .zsiq_float, iframe[src*="zoho"] {
          display:none !important;
        }
      `;
        document.head.appendChild(style);
      }

      function ipLookUp() {
        fetch(
          "https://asia-east2-handy-amplifier-259811.cloudfunctions.net/geolocation",
        )
          .then((res) => res.json())
          .then((data) => {
            if (data.country === "IN") {
              blockPage();
            }
          })
          .catch(() => {
            // fail silently (don’t block UX)
          });
      }

      // ✅ Delay execution (NOT in critical path)
      window.addEventListener("load", () => {
        setTimeout(ipLookUp, 1000); // delay = big performance win
      });
    </script>
