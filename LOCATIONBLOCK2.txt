    <script>
      window.$zoho = window.$zoho || {};
      $zoho.salesiq = $zoho.salesiq || {};

      let geoChecked = false;

      function blockPage() {
        document.title = "Page not found";

        const blocker = document.createElement("div");
        blocker.innerHTML = `
      <div style="
        position:fixed;
        inset:0;
        display:flex;
        align-items:center;
        justify-content:center;
        background:#f3f4f6;
        z-index:999999;
        font-family:sans-serif;
      ">
        <div style="
          background:#fff;
          padding:32px;
          border-radius:16px;
          box-shadow:0 10px 30px rgba(0,0,0,0.1);
          text-align:center;
          max-width:400px;
          width:90%;
        ">
          <h1 style="font-size:64px;font-weight:900;color:#dc2626;margin-bottom:8px;">
            404
          </h1>
          <h2 style="font-size:24px;font-weight:700;">
            Page not found
          </h2>
        </div>
      </div>
    `;

        document.body.appendChild(blocker);

        // hide Zoho chat if present
        const style = document.createElement("style");
        style.innerHTML = `
      #zsiq_float, .zsiq_float, iframe[src*="zoho"] {
        display:none !important;
      }
    `;
        document.head.appendChild(style);

        document.body.style.overflow = "hidden";
      }

      function ipLookUp() {
        fetch(
          "https://asia-east2-handy-amplifier-259811.cloudfunctions.net/geolocation",
        )
          .then((res) => res.json())
          .then((data) => {
            if (data.country === "IN") {
              blockPage();
            }
          })
          .catch(() => {
            // fail silently (don’t affect UX)
          });
      }

      // 🚀 Optimized execution (no impact on LCP)
      window.addEventListener("load", () => {
        if ("requestIdleCallback" in window) {
          requestIdleCallback(() => {
            if (!geoChecked) {
              geoChecked = true;
              ipLookUp();
            }
          });
        } else {
          setTimeout(() => {
            if (!geoChecked) {
              geoChecked = true;
              ipLookUp();
            }
          }, 1500);
        }
      });
    </script>
